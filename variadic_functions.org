* VARIADIC FUNCTIONS
A variadic function accepts a variable number of arguments.
** LINKS
[[https://en.wikipedia.org/wiki/Variadic_function][Wikipedia]]
** PROGRAMMING LANGUAGES
*** C
#+begin_src c  
#include <stdarg.h>
#include <stdlib.h>
#include <stdio.h>

double mean(int count,...){
  va_list ap;
  va_start(ap, count);
  
  double sum = 0;

  for(size_t i = 0; i < count; i++){
    sum += va_arg(ap, double);
  }
  va_end(ap);
  return sum / count;
}
    
int main (int argc, char** argv) {
  printf ("%f\n", mean(3, 10.5, 25.4, 30.5));
  printf ("%f\n", mean(5, 1.0, 21.0, 3.0, 9.0, 19.0));
  return EXIT_SUCCESS;
}
#+end_src
