* Circle Mathematics
** FORMULA
*** Area
A = πr²
*** Circumference
C = 2πr
*** Radius
r = √(A ÷ π)

** C
#+begin_src c
  #include <math.h>
  
  double circle_area(double radius){
    return M_PI * pow(radius,2);
  }
  
  double circle_circumference(double radius){
    return 2 * M_PI * radius;
  } 
#+end_src

#+end_src
